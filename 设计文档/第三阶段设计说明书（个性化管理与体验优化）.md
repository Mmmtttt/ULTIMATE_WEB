### 2.3 批量操作设计
```python
class BatchProcessor:
    """批量处理器"""
    
    def __init__(self):
        self.batch_size = 50  # 每批处理数量
        self.retry_times = 3  # 重试次数
    
    def batch_update_comics(self, comic_ids, update_func):
        """
        批量更新漫画
        
        策略：
        1. 分批处理，避免内存溢出
        2. 单批失败不影响其他批
        3. 记录失败项，支持重试
        """
        results = {
            'success': [],
            'failed': []
        }
        
        for batch in self._split_batches(comic_ids, self.batch_size):
            try:
                for comic_id in batch:
                    update_func(comic_id)
                    results['success'].append(comic_id)
            except Exception as e:
                results['failed'].extend(batch)
                logger.error(f"批量处理失败: {e}")
        
        return results
    
    def _split_batches(self, items, batch_size):
        """分批"""
        for i in range(0, len(items), batch_size):
            yield items[i:i + batch_size]
```

```python
class ConfigManager:
    """配置管理器（单例模式）"""
    
    _instance = None
    
    # 默认配置
    DEFAULT_CONFIG = {
        'preload_num': 3,
        'default_page_mode': 'left_right',
        'default_background': 'white',
        'auto_hide_toolbar': True,
        'show_page_number': True
    }
    
    # 配置校验规则
    VALIDATION_RULES = {
        'preload_num': {
            'type': int,
            'min': 1,
            'max': 10
        },
        'default_page_mode': {
            'type': str,
            'choices': ['left_right', 'up_down']
        },
        'default_background': {
            'type': str,
            'choices': ['white', 'dark', 'eye_protection']
        }
    }
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._config = {}
            cls._instance._load_config()
        return cls._instance
    
    def get(self, key, default=None):
        """获取配置项"""
        return self._config.get(key, default or self.DEFAULT_CONFIG.get(key))
    
    def set(self, key, value):
        """
        设置配置项
        
        流程：
        1. 校验值类型和范围
        2. 更新内存配置
        3. 持久化到JSON
        4. 触发变更事件
        """
        if not self._validate(key, value):
            raise ValueError(f"配置项{key}的值{value}不合法")
        
        old_value = self._config.get(key)
        self._config[key] = value
        
        try:
            self._save_config()
            self._emit_change(key, old_value, value)
        except Exception as e:
            # 回滚
            self._config[key] = old_value
            raise e
    
    def reset(self):
        """重置为默认配置"""
        self._config = self.DEFAULT_CONFIG.copy()
        self._save_config()
    
    def _validate(self, key, value):
        """校验配置值"""
        rule = self.VALIDATION_RULES.get(key)
        if not rule:
            return True
        
        if not isinstance(value, rule['type']):
            return False
        
        if 'min' in rule and value < rule['min']:
            return False
        
        if 'max' in rule and value > rule['max']:
            return False
        
        if 'choices' in rule and value not in rule['choices']:
            return False
        
        return True
    
    def _emit_change(self, key, old_value, new_value):
        """触发配置变更事件"""
        # 通知所有监听者
        pass
```

### 4.2 ReaderEngine核心设计
```javascript
// composables/useReaderEngine.js
export function useReaderEngine(comicId, totalPage) {
  // ============ 状态 ============
  const state = ref('IDLE'); // IDLE, LOADING, READING, ZOOMING, MENU
  const currentPage = ref(1);
  const pageMode = ref('left_right'); // left_right, up_down
  const background = ref('white');
  const scale = ref(1);
  const position = ref({ x: 0, y: 0 });
  
  // ============ 配置 ============
  const config = useConfigStore();
  const preloadCount = computed(() => config.preloadCount);
  
  // ============ 缓存 ============
  const imageCache = ref(new Map()); // 图片缓存
  const preloadedPages = ref(new Set()); // 已预加载页码
  
  // ============ 加载 ============
  const loadComic = async (page = 1) => {
    state.value = 'LOADING';
    
    try {
      // 加载阅读进度
      await loadProgress();
      
      // 跳转到指定页
      currentPage.value = page;
      
      // 预加载周围图片
      preloadRange();
      
      state.value = 'READING';
    } catch (e) {
      state.value = 'IDLE';
      throw e;
    }
  };
  
  // ============ 翻页 ============
  const nextPage = async () => {
    if (state.value !== 'READING') return;
    if (currentPage.value >= totalPage) return;
    
    currentPage.value++;
    await saveProgress();
    preloadRange();
  };
  
  const prevPage = async () => {
    if (state.value !== 'READING') return;
    if (currentPage.value <= 1) return;
    
    currentPage.value--;
    await saveProgress();
    preloadRange();
  };
  
  const jumpToPage = async (page) => {
    if (page < 1 || page > totalPage) {
      showToast(`页码范围1-${totalPage}`);
      return;
    }
    
    currentPage.value = page;
    await saveProgress();
    preloadRange();
  };
  
  // ============ 预加载 ============
  const preloadRange = () => {
    const range = preloadCount.value;
    const pages = [];
    
    for (let i = -range; i <= range; i++) {
      const page = currentPage.value + i;
      if (page >= 1 && page <= totalPage && !preloadedPages.value.has(page)) {
        pages.push(page);
      }
    }
    
    // 批量预加载
    pages.forEach(page => preloadImage(page));
  };
  
  const preloadImage = (page) => {
    if (preloadedPages.value.has(page)) return;
    
    const img = new Image();
    img.src = getImageUrl(comicId, page);
    img.onload = () => {
      imageCache.value.set(page, img);
      preloadedPages.value.add(page);
    };
  };
  
  // ============ 缩放 ============
  const zoomIn = () => {
    scale.value = Math.min(2.0, scale.value + 0.1);
    state.value = 'ZOOMING';
  };
  
  const zoomOut = () => {
    scale.value = Math.max(0.5, scale.value - 0.1);
    state.value = 'ZOOMING';
  };
  
  const resetZoom = () => {
    scale.value = 1;
    position.value = { x: 0, y: 0 };
    state.value = 'READING';
  };
  
  // ============ 模式切换 ============
  const switchPageMode = async () => {
    pageMode.value = pageMode.value === 'left_right' ? 'up_down' : 'left_right';
    await saveProgress();
    
    // 重新加载以应用新模式
    await loadComic(currentPage.value);
  };
  
  const switchBackground = async () => {
    const colors = ['white', 'dark', 'eye_protection'];
    const currentIndex = colors.indexOf(background.value);
    background.value = colors[(currentIndex + 1) % colors.length];
    await saveProgress();
  };
  
  // ============ 进度保存 ============
  const saveProgress = async () => {
    const progress = {
      comic_id: comicId,
      current_page: currentPage.value,
      page_mode: pageMode.value,
      background: background.value,
      scale: scale.value
    };
    
    await api.comic.saveProgress(progress);
  };
  
  const loadProgress = async () => {
    const res = await api.comic.getProgress(comicId);
    if (res.code === 200 && res.data) {
      const data = res.data;
      currentPage.value = data.current_page || 1;
      pageMode.value = data.page_mode || config.defaultPageMode;
      background.value = data.background || config.defaultBackground;
      scale.value = data.scale || 1;
    }
  };
  
  // ============ 快捷键 ============
  const handleKeydown = (e) => {
    const keyMap = {
      'ArrowLeft': prevPage,
      'ArrowRight': nextPage,
      'a': prevPage,
      'd': nextPage,
      ' ': nextPage,
      'ArrowUp': () => { /* 上滑 */ },
      'ArrowDown': () => { /* 下滑 */ },
      'w': () => { /* 上滑 */ },
      's': () => { /* 下滑 */ },
      '+': zoomIn,
      '=': zoomIn,
      '-': zoomOut,
      '_': zoomOut,
      '0': resetZoom,
      'Escape': () => { /* 退出 */ },
      'b': switchBackground,
      'm': switchPageMode
    };
    
    const handler = keyMap[e.key];
    if (handler) {
      e.preventDefault();
      handler();
    }
  };
  
  // ============ 手势 ============
  const handleTouchStart = (e) => {
    // 记录触摸起始位置
  };
  
  const handleTouchMove = (e) => {
    // 处理滑动、捏合
  };
  
  const handleTouchEnd = (e) => {
    // 判断滑动方向，执行翻页
  };
  
  // ============ 生命周期 ============
  onMounted(() => {
    window.addEventListener('keydown', handleKeydown);
    loadComic();
  });
  
  onUnmounted(() => {
    window.removeEventListener('keydown', handleKeydown);
    saveProgress();
  });
  
  return {
    state,
    currentPage,
    pageMode,
    background,
    scale,
    position,
    nextPage,
    prevPage,
    jumpToPage,
    zoomIn,
    zoomOut,
    resetZoom,
    switchPageMode,
    switchBackground,
    handleTouchStart,
    handleTouchMove,
    handleTouchEnd
  };
}
```

```vue
  const pages = [];
  for (let i = 0; i < visibleCount.value; i++) {
    const page = startIndex.value + i + 1;
    if (page <= props.totalPage) {
      pages.push(page);
    }
  }
  return pages;
});

const offsetY = computed(() => startIndex.value * pageHeight.value);

// 方法
const onScroll = () => {
  const scrollTop = container.value.scrollTop;
  startIndex.value = Math.floor(scrollTop / pageHeight.value);
  
  // 更新当前页码
  const centerIndex = startIndex.value + Math.floor(visibleCount.value / 2);
  currentPage.value = Math.min(centerIndex + 1, props.totalPage);
  
  // 预加载
  preloadRange();
};

const shouldRender = (page) => {
  // 只渲染已加载的图片和可视区域内的图片
  return loadedPages.value.has(page) || isNearViewport(page);
};

const isNearViewport = (page) => {
  const index = page - 1;
  return Math.abs(index - startIndex.value) <= 2;
};

const preloadRange = () => {
  const range = 3;
  for (let i = -range; i <= range; i++) {
    const page = currentPage.value + i;
    if (page >= 1 && page <= props.totalPage) {
      preloadImage(page);
    }
  }
};

const preloadImage = (page) => {
  if (loadedPages.value.has(page)) return;
  
  const img = new Image();
  img.src = getImageUrl(props.comicId, page);
  img.onload = () => {
    loadedPages.value.add(page);
    // 更新实际高度
    if (img.naturalHeight > 0) {
      pageHeight.value = img.naturalHeight;
    }
  };
};
</script>
```

```javascript
// composables/useGesture.js
export function useGesture(element) {
  const gesture = ref({
    type: null, // tap, longPress, swipe, pinch
    startX: 0,
    startY: 0,
    startTime: 0,
    startDistance: 0
  });
  
  const emit = defineEmits(['tap', 'longPress', 'swipeLeft', 'swipeRight', 'swipeUp', 'swipeDown', 'pinch']);
  
  const onTouchStart = (e) => {
    const touch = e.touches[0];
    gesture.value = {
      type: null,
      startX: touch.clientX,
      startY: touch.clientY,
      startTime: Date.now(),
      startDistance: getDistance(e.touches)
    };
  };
  
  const onTouchMove = (e) => {
    if (e.touches.length === 2) {
      // 双指缩放
      const distance = getDistance(e.touches);
      const scale = distance / gesture.value.startDistance;
      emit('pinch', { scale });
    }
  };
  
  const onTouchEnd = (e) => {
    const touch = e.changedTouches[0];
    const deltaX = touch.clientX - gesture.value.startX;
    const deltaY = touch.clientY - gesture.value.startY;
    const deltaTime = Date.now() - gesture.value.startTime;
    
    // 判断手势类型
    if (deltaTime < 300 && Math.abs(deltaX) < 10 && Math.abs(deltaY) < 10) {
      // 点击
      emit('tap', { x: touch.clientX, y: touch.clientY });
    } else if (deltaTime > 500 && Math.abs(deltaX) < 10 && Math.abs(deltaY) < 10) {
      // 长按
      emit('longPress', { x: touch.clientX, y: touch.clientY });
    } else if (Math.abs(deltaX) > Math.abs(deltaY) && Math.abs(deltaX) > 50) {
      // 水平滑动
      if (deltaX > 0) {
        emit('swipeRight');
      } else {
        emit('swipeLeft');
      }
    } else if (Math.abs(deltaY) > 50) {
      // 垂直滑动
      if (deltaY > 0) {
        emit('swipeDown');
      } else {
        emit('swipeUp');
      }
    }
  };
  
  const getDistance = (touches) => {
    const dx = touches[0].clientX - touches[1].clientX;
    const dy = touches[0].clientY - touches[1].clientY;
    return Math.sqrt(dx * dx + dy * dy);
  };
  
  onMounted(() => {
    element.value.addEventListener('touchstart', onTouchStart);
    element.value.addEventListener('touchmove', onTouchMove);
    element.value.addEventListener('touchend', onTouchEnd);
  });
  
  onUnmounted(() => {
    element.value.removeEventListener('touchstart', onTouchStart);
    element.value.removeEventListener('touchmove', onTouchMove);
    element.value.removeEventListener('touchend', onTouchEnd);
  });
  
  return { gesture };
}
```

```vue
    emit('back');

  showIndicator.value = false;
```

```javascript
// utils/imageCache.js
class ImageCache {
  constructor(maxSize = 50) {
    this.cache = new Map();
    this.maxSize = maxSize;
    this.accessOrder = [];
  }
  
  get(key) {
    if (this.cache.has(key)) {
      // 更新访问顺序
      this.updateAccessOrder(key);
      return this.cache.get(key);
    }
    return null;
  }
  
  set(key, value) {
    if (this.cache.size >= this.maxSize) {
      // LRU淘汰
      const oldestKey = this.accessOrder.shift();
      this.cache.delete(oldestKey);
    }
    
    this.cache.set(key, value);
    this.accessOrder.push(key);
  }
  
  updateAccessOrder(key) {
    const index = this.accessOrder.indexOf(key);
    if (index > -1) {
      this.accessOrder.splice(index, 1);
      this.accessOrder.push(key);
    }
  }
  
  clear() {
    this.cache.clear();
    this.accessOrder = [];
  }
}

export const imageCache = new ImageCache(50);
```

```javascript
    this.marks[label] = performance.now();
  },
  
  // 结束计时
  end(label) {
    if (this.marks[label]) {
      const duration = performance.now() - this.marks[label];

      delete this.marks[label];
      return duration;
    }
    return 0;
  },
  
  // 测量函数执行时间

      this.start(label);
      const result = fn(...args);
      this.end(label);
      return result;
    };

  // 上报性能数据（可选）

      memory: performance.memory?.usedJSHeapSize,
      timing: performance.timing,
```

|------|----------|
| 长漫画内存溢出 | 虚拟滚动，只渲染可视区域 |

| 频繁操作卡顿 | 防抖节流，异步处理 |

| 触屏手势冲突 | touch-action控制 |
| 横竖屏切换问题 | 监听orientationchange |
| 不同尺寸适配 | 响应式设计，rem/vw |
