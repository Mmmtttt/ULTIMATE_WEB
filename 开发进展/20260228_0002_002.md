## 核心代码片段

### 新 Store 使用示例
```javascript
// 从新的 stores 目录导入
import { 
  useComicStore, 
  useCacheStore, 
  useConfigStore,
  useReaderStore,
  useTagStore 
} from '@/stores'

// 使用缓存 Store
const cacheStore = useCacheStore()
const cached = cacheStore.getListCache()
if (cached) {
  return cached
}
cacheStore.setListCache(data)

// 使用配置 Store
const configStore = useConfigStore()
configStore.setPreloadNum(5)
configStore.togglePageMode()
configStore.toggleBackground()

// 使用阅读器 Store
const readerStore = useReaderStore()
readerStore.initReader('comic_001', 100, 1)
readerStore.jumpToPage(10)
readerStore.nextPage()
readerStore.zoomIn()

// 使用标签 Store
const tagStore = useTagStore()
tagStore.fetchTags()
tagStore.addTag('新标签')
tagStore.selectTag('tag_001')

// 使用漫画 Store
const comicStore = useComicStore()
comicStore.fetchComics()
comicStore.fetchComicDetail('comic_001')
comicStore.updateScore('comic_001', 8.5)
```

### 旧 Store 仍然可用
```javascript
// 原有的导入方式仍然有效
import { useComicStore } from '@/store/modules/comic'

const store = useComicStore()
store.fetchComics()  // 正常工作
```
