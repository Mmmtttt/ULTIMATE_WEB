## 核心代码片段

### 使用新的 Composables
```javascript
// 从 composables 目录导入
import { 
  useReader, 
  useComic, 
  useSearch, 
  useCache 
} from '@/composables'

// 使用阅读器组合
const {
  currentPage,
  totalPage,
  progress,
  nextPage,
  prevPage,
  jumpToPage,
  zoomIn,
  zoomOut,
  preloadImages
} = useReader()

// 使用漫画组合
const {
  comics,
  currentComic,
  fetchComics,
  fetchDetail,
  updateScore,
  selectComic,
  toggleSelection
} = useComic()

// 使用搜索组合
const {
  keyword,
  includeTags,
  excludeTags,
  search,
  debouncedSearch,
  filterByTags,
  clearAllFilters
} = useSearch()

// 使用缓存组合
const {
  stats,
  isListCacheValid,
  clearAllCache,
  clearDetailCache,
  getCacheSize
} = useCache()
```
