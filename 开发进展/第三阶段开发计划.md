### 4.2 图床代理技术方案

#### 实现原理
```
图床代理：
1. 后端提供图片代理接口
2. 前端请求后端代理接口
3. 后端从图床下载图片
4. 后端缓存图片到本地
5. 返回图片数据给前端
```

#### 核心代码
```python
# infrastructure/storage/image_host_proxy.py
class ImageHostProxy:
    def __init__(self, cache_dir: str = "./cache/image_host"):
        self._cache_dir = cache_dir
        self._session = requests.Session()
        self._session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        })
    
    def get_image(self, comic_id: str, page_num: int):
        """获取图片（带缓存）"""
        # 检查本地缓存
        cache_path = self._get_cache_path(comic_id, page_num)
        if os.path.exists(cache_path):
            return open(cache_path, 'rb')
        
        # 获取图床 URL
        url = self._get_image_url(comic_id, page_num)
        if not url:
            return None
        
        # 从图床下载
        try:
            response = self._session.get(url, timeout=30, stream=True)
            response.raise_for_status()
            
            # 保存到缓存
            os.makedirs(os.path.dirname(cache_path), exist_ok=True)
            with open(cache_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)
            
            return open(cache_path, 'rb')
        except Exception as e:
            logger.error(f"下载图床图片失败: {url}, 错误: {e}")
            return None
    
    def _get_image_url(self, comic_id: str, page_num: int) -> Optional[str]:
        """从配置中获取图床 URL"""
        data = self._storage.read()
        cache = data.get("image_host_cache", {})
        urls = cache.get(comic_id, [])
        
        if 1 <= page_num <= len(urls):
            return urls[page_num - 1]
        return None
```

### 4.3 导入策略技术方案

#### 实现原理
```
导入策略模式：
1. 定义导入策略基类
2. 每种导入方式实现策略接口
3. 工厂模式创建策略实例
4. 应用服务协调导入流程
5. 支持进度回调和错误处理
```

#### 核心代码
```python
# domain/import_strategy/base.py
from abc import ABC, abstractmethod
from typing import Dict, Any, Callable
from domain.comic.entity import Comic

class ImportStrategy(ABC):
    """导入策略基类"""
    
    @abstractmethod
    def validate(self, params: Dict[str, Any]) -> bool:
        """验证导入参数"""
        pass
    
    @abstractmethod
    def execute(
        self,
        params: Dict[str, Any],
        progress_callback: Callable[[int], None] = None
    ) -> Comic:
        """执行导入"""
        pass
    
    @abstractmethod
    def get_name(self) -> str:
        """获取策略名称"""
        pass

# application/import_app_service.py
class ImportAppService:
    """导入应用服务"""
    
    def __init__(
        self,
        comic_repository: ComicRepository,
        image_processor: ImageProcessor
    ):
        self._comic_repo = comic_repository
        self._image_processor = image_processor
        self._factory = ImportStrategyFactory()
    
    def import_comic(
        self,
        strategy_type: str,
        params: Dict[str, Any],
        progress_callback: Callable[[int], None] = None
    ) -> Result[Comic]:
        """导入漫画"""
        # 创建策略
        strategy = self._factory.create(
            strategy_type,
            image_processor=self._image_processor
        )
        
        # 验证参数
        if not strategy.validate(params):
            return Result.fail("导入参数验证失败")
        
        try:
            # 执行导入
            comic = strategy.execute(params, progress_callback)
            
            # 保存漫画
            self._comic_repo.save(comic)
            
            # 记录导入历史
            self._record_import_history(strategy_type, comic.id, "success")
            
            return Result.ok(comic)
        except Exception as e:
            self._record_import_history(strategy_type, params.get("comic_id"), "failed", str(e))
            return Result.fail(f"导入失败: {str(e)}")
```
