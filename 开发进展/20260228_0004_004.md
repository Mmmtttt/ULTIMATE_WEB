## 核心代码片段

### 使用新组件
```vue
<template>
  <div class="reader">
    <!-- 工具栏 -->
    <ReaderToolbar
      :visible="toolbarVisible"
      :title="comic.title"
      :current-page="currentPage"
      :total-page="totalPage"
      :can-go-prev="canGoPrev"
      :can-go-next="canGoNext"
      :scale-percent="scalePercent"
      @back="goBack"
      @prev="prevPage"
      @next="nextPage"
      @zoom-in="zoomIn"
      @zoom-out="zoomOut"
    />
    
    <!-- 设置面板 -->
    <ReaderSettings
      v-model:visible="settingsVisible"
      v-model:mode="pageMode"
      v-model:background="background"
      v-model:preload-num="preloadNum"
    />
    
    <!-- 漫画网格 -->
    <ComicGrid
      :comics="comics"
      :selected-ids="selectedIds"
      :selectable="isSelecting"
      @card-click="handleCardClick"
      @toggle-select="handleToggleSelect"
    />
    
    <!-- 标签筛选 -->
    <TagFilter
      v-model:include-tags="includeTags"
      v-model:exclude-tags="excludeTags"
      :tags="tags"
      show-count
    />
  </div>
</template>

<script setup>
import {
  ReaderToolbar,
  ReaderProgress,
  ReaderSettings,
  ComicGrid,
  TagFilter
} from '@/components'
</script>
```
