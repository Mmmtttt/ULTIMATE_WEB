## 1. 重构背景与目标

### 1.1 当前问题分析

经过多轮迭代开发，项目存在以下架构问题：

#### 1.1.1 后端问题
- **职责混杂**：Routes 层包含业务逻辑，Service 层边界不清晰
- **重复代码**：JSON 读写、错误处理等逻辑分散在各处
- **紧耦合**：Utils 层直接操作数据，缺乏抽象层
- **扩展困难**：新增功能需要修改多处代码（霰弹式修改）
- **缺乏统一规范**：返回格式、日志记录、参数校验不统一

#### 1.1.2 前端问题
- **组件臃肿**：ComicReader.vue 超过 600 行，职责过多
- **状态管理混乱**：缓存逻辑与业务逻辑混合
- **重复逻辑**：API 调用、错误处理在多个组件重复
- **缺乏抽象**：没有通用的列表、表单、弹窗组件
- **硬编码过多**：配置项、常量分散在代码中

### 1.2 重构目标

1. **高内聚低耦合**：每个模块职责单一，依赖清晰
2. **便于扩展**：新增功能只需添加代码，无需修改现有代码
3. **可维护性**：代码结构清晰，易于理解和修改
4. **支持未来需求**：
   - Python 脚本下载漫画
   - 本地目录导入
   - ZIP 文件上传导入
   - 图床链接支持

```python
# infrastructure/common/result.py
from typing import Generic, TypeVar, Optional
from enum import Enum

T = TypeVar('T')

class ResultCode(Enum):
    SUCCESS = 200
    BAD_REQUEST = 400
    NOT_FOUND = 404
    INTERNAL_ERROR = 500

class Result(Generic[T]):
    """统一返回结果"""
    
    def __init__(
        self,
        code: ResultCode = ResultCode.SUCCESS,
        data: Optional[T] = None,
        message: str = "成功"
    ):
        self.code = code
        self.data = data
        self.message = message
    
    @property
    def is_success(self) -> bool:
        return self.code == ResultCode.SUCCESS
    
    @staticmethod
    def ok(data: T = None, message: str = "成功") -> "Result[T]":
        return Result(ResultCode.SUCCESS, data, message)
    
    @staticmethod
    def fail(message: str, code: ResultCode = ResultCode.BAD_REQUEST) -> "Result":
        return Result(code, None, message)
    
    def to_dict(self) -> dict:
        return {
            "code": self.code.value,
            "data": self.data,
            "message": self.message
        }
```

```python
# infrastructure/persistence/json_repository.py
from domain.comic.repository import ComicRepository
from domain.comic.entity import Comic
from infrastructure.persistence.json_storage import JsonStorage

class ComicJsonRepository(ComicRepository):
    """漫画 JSON 仓储实现"""
    
    def __init__(self, storage: JsonStorage):
        self._storage = storage
    
    def get_by_id(self, comic_id: str) -> Optional[Comic]:
        data = self._storage.read()
        comics = data.get("comics", [])
        comic_data = next(
            (c for c in comics if c["id"] == comic_id),
            None
        )
        return Comic.from_dict(comic_data) if comic_data else None
    
    def get_all(self, sort_by: str = None) -> List[Comic]:
        data = self._storage.read()
        comics = [Comic.from_dict(c) for c in data.get("comics", [])]
        
        if sort_by:
            comics = self._sort_comics(comics, sort_by)
        
        return comics
    
    def save(self, comic: Comic) -> None:
        data = self._storage.read()
        comics = data.get("comics", [])
        
        # 查找并更新或追加
        index = next(
            (i for i, c in enumerate(comics) if c["id"] == comic.id),
            -1
        )
        
        if index >= 0:
            comics[index] = comic.to_dict()
        else:
            comics.append(comic.to_dict())
        
        data["comics"] = comics
        self._storage.write(data)
    
    def _sort_comics(self, comics: List[Comic], sort_by: str) -> List[Comic]:
        """排序漫画"""
        sort_key = {
            "create_time": lambda c: c.create_time,
            "score": lambda c: c.score or 0,
            "read_time": lambda c: c.last_read_time or ""
        }.get(sort_by)
        
        if sort_key:
            comics = sorted(comics, key=sort_key, reverse=True)
        
        return comics
```

```python
    def validate(self, params: Dict[str, Any]) -> bool:

        pass

    def execute(self, params: Dict[str, Any]) -> Comic:
        """执行导入"""

    @abstractmethod
    def get_name(self) -> str:
        """获取策略名称"""

import subprocess
from .base import ImportStrategy
from domain.comic.entity import Comic

class ScriptImportStrategy(ImportStrategy):
    """Python 脚本导入策略"""
    
    def validate(self, params: Dict[str, Any]) -> bool:
        required = ["script_path", "comic_id"]
        return all(key in params for key in required)
    
    def execute(self, params: Dict[str, Any]) -> Comic:
        script_path = params["script_path"]
        comic_id = params["comic_id"]
        
        # 执行 Python 脚本下载
        result = subprocess.run(
            ["python", script_path, comic_id],
            capture_output=True,
            text=True
        )
        
        if result.returncode != 0:
            raise ImportError(f"脚本执行失败: {result.stderr}")
        
        # 解析脚本输出，创建漫画实体
        return self._parse_script_output(result.stdout)
    
    def get_name(self) -> str:
        return "script_import"

import shutil

class DirectoryImportStrategy(ImportStrategy):
    """本地目录导入策略"""

        self._image_processor = image_processor

        required = ["source_dir", "comic_id", "title"]
        if not all(key in params for key in required):
            return False
        
        source_dir = params["source_dir"]
        return os.path.exists(source_dir) and os.path.isdir(source_dir)
    
    def execute(self, params: Dict[str, Any]) -> Comic:
        source_dir = params["source_dir"]
        comic_id = params["comic_id"]
        title = params["title"]
        
        # 扫描图片文件
        image_files = self._scan_images(source_dir)
        
        if not image_files:
            raise ImportError("目录中没有找到图片文件")
        
        # 复制到目标目录
        target_dir = self._get_target_dir(comic_id)
        self._copy_images(image_files, target_dir)
        
        # 生成封面

            image_files[0]
        )

            id=comic_id,
            title=title,
            total_page=len(image_files),
            cover_path=cover_path,
            create_time=datetime.now().isoformat()
        )
    
    def get_name(self) -> str:

                "comic_id": comic_id,
                "title": title
            })

        return "zip_import"

from .base import ImportStrategy
from domain.comic.entity import Comic
from infrastructure.storage.image_host_proxy import ImageHostProxy

class ImageHostImportStrategy(ImportStrategy):
    """图床链接导入策略"""
    
    def __init__(self, image_host_proxy: ImageHostProxy):
        self._proxy = image_host_proxy
    
    def validate(self, params: Dict[str, Any]) -> bool:
        required = ["comic_id", "title", "image_urls"]
        return all(key in params for key in required)
    
    def execute(self, params: Dict[str, Any]) -> Comic:
        comic_id = params["comic_id"]
        title = params["title"]
        image_urls = params["image_urls"]
        
        # 验证图床链接
        valid_urls = []
        for url in image_urls:
            if self._proxy.validate_url(url):
                valid_urls.append(url)
        
        if not valid_urls:
            raise ImportError("没有有效的图床链接")
        
        # 保存图床配置
        self._proxy.save_comic_images(comic_id, valid_urls)
        
        # 创建漫画实体（图床模式）
        return Comic(
            id=comic_id,
            title=title,
            total_page=len(valid_urls),
            cover_path=valid_urls[0],  # 第一张图作为封面
            image_source="image_host",  # 标记为图床来源
            create_time=datetime.now().isoformat()
        )
    
    def get_name(self) -> str:
        return "image_host_import"

class ImportStrategyFactory:

        "script": ScriptImportStrategy,
        "directory": DirectoryImportStrategy,
        "zip": ZipImportStrategy,
        "image_host": ImageHostImportStrategy
    }
    
    @classmethod
    def create(cls, strategy_type: str, **deps) -> ImportStrategy:

        strategy_class = cls._strategies.get(strategy_type)
        if not strategy_class:

        return strategy_class(**deps)
    
    @classmethod
    def register(cls, strategy_type: str, strategy_class: Type[ImportStrategy]):
        """注册新策略"""
        cls._strategies[strategy_type] = strategy_class
```

```python
# infrastructure/storage/image_host_proxy.py
from typing import List, Optional
import requests
from urllib.parse import urlparse

class ImageHostProxy:
    """图床代理服务"""
    
    def __init__(self, cache_dir: str = "./cache/image_host"):
        self._cache_dir = cache_dir
        self._comic_images: Dict[str, List[str]] = {}
    
    def validate_url(self, url: str) -> bool:
        """验证图床链接是否有效"""
        try:
            response = requests.head(url, timeout=5, allow_redirects=True)
            return response.status_code == 200
        except:
            return False
    
    def save_comic_images(self, comic_id: str, image_urls: List[str]):
        """保存漫画的图床链接"""
        self._comic_images[comic_id] = image_urls
        # 持久化到 JSON
        self._persist()
    
    def get_image_url(self, comic_id: str, page_num: int) -> Optional[str]:
        """获取指定页码的图片链接"""
        urls = self._comic_images.get(comic_id, [])
        if 1 <= page_num <= len(urls):
            return urls[page_num - 1]
        return None
    
    def get_image_stream(self, comic_id: str, page_num: int):
        """获取图片流（支持缓存）"""
        url = self.get_image_url(comic_id, page_num)
        if not url:
            return None
        
        # 检查本地缓存
        cache_path = self._get_cache_path(comic_id, page_num)
        if os.path.exists(cache_path):
            return open(cache_path, 'rb')
        
        # 从图床下载
        try:
            response = requests.get(url, timeout=30, stream=True)
            response.raise_for_status()
            
            # 保存到缓存
            os.makedirs(os.path.dirname(cache_path), exist_ok=True)
            with open(cache_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)
            
            return open(cache_path, 'rb')
        except Exception as e:
            logger.error(f"下载图床图片失败: {url}, 错误: {e}")
            return None
    
    def _get_cache_path(self, comic_id: str, page_num: int) -> str:
        """获取缓存路径"""
        return os.path.join(self._cache_dir, comic_id, f"{page_num}.jpg")

        self._local = local_storage
        self._proxy = image_host_proxy
    
    def get_image(self, comic_id: str, page_num: int, source: str = "local"):
        """获取图片（自动判断来源）"""
        if source == "image_host":
            return self._proxy.get_image_stream(comic_id, page_num)
        else:
            return self._local.get_image(comic_id, page_num)
    
    def get_cover(self, comic_id: str, cover_path: str):
        """获取封面"""
        if cover_path.startswith("http"):
            # 图床封面
            return self._proxy.get_image_by_url(cover_path)
        else:
            # 本地封面
            return self._local.get_file(cover_path)
```

#### 3.3.1 阅读器逻辑抽离

```javascript
// composables/useReader.js
import { ref, computed, watch } from 'vue'
import { useReaderStore } from '@/stores/reader'
import { useConfigStore } from '@/stores/config'

export function useReader(comicId, totalPage) {
  const readerStore = useReaderStore()
  const configStore = useConfigStore()
  
  // 状态
  const state = ref('IDLE') // IDLE, LOADING, READING, ZOOMING, MENU
  const currentPage = ref(1)
  const scale = ref(1)
  const position = ref({ x: 0, y: 0 })
  
  // 配置
  const pageMode = computed(() => configStore.pageMode)
  const background = computed(() => configStore.background)
  const preloadCount = computed(() => configStore.preloadCount)
  
  // 计算属性
  const progress = computed(() => ({
    current: currentPage.value,
    total: totalPage,
    percentage: Math.round((currentPage.value / totalPage) * 100)
  }))
  
  const canGoNext = computed(() => currentPage.value < totalPage)
  const canGoPrev = computed(() => currentPage.value > 1)
  
  // 方法
  const loadComic = async (page = 1) => {
    state.value = 'LOADING'
    
    try {
      // 加载阅读进度
      const savedProgress = await readerStore.loadProgress(comicId)
      if (savedProgress) {
        currentPage.value = savedProgress.page
        // 恢复其他状态
        if (savedProgress.mode) configStore.setPageMode(savedProgress.mode)
        if (savedProgress.background) configStore.setBackground(savedProgress.background)
        if (savedProgress.scale) scale.value = savedProgress.scale
      } else {
        currentPage.value = page
      }
      
      // 预加载
      preloadRange()
      
      state.value = 'READING'
    } catch (error) {
      state.value = 'IDLE'
      throw error
    }
  }
  
  const nextPage = async () => {
    if (!canGoNext.value || state.value !== 'READING') return
    
    currentPage.value++
    await saveProgress()
    preloadRange()
  }
  
  const prevPage = async () => {
    if (!canGoPrev.value || state.value !== 'READING') return
    
    currentPage.value--
    await saveProgress()
    preloadRange()
  }
  
  const jumpToPage = async (page) => {
    if (page < 1 || page > totalPage) {
      showToast(`页码范围 1-${totalPage}`)
      return
    }
    
    currentPage.value = page
    await saveProgress()
    preloadRange()
  }
  
  const saveProgress = async () => {
    await readerStore.saveProgress(comicId, {
      page: currentPage.value,
      mode: pageMode.value,
      background: background.value,
      scale: scale.value
    })
  }
  
  // 预加载
  const preloadRange = () => {
    const range = preloadCount.value
    const pages = []
    
    // 向后加载
    for (let i = 0; i <= range; i++) {
      const page = currentPage.value + i
      if (page <= totalPage) pages.push(page)
    }
    
    // 向前加载 1 张
    const prevPage = currentPage.value - 1
    if (prevPage >= 1) pages.push(prevPage)
    
    // 去重并预加载
    [...new Set(pages)].forEach(page => {
      readerStore.preloadImage(comicId, page)
    })
  }
  
  // 缩放
  const zoomIn = () => {
    scale.value = Math.min(2.0, scale.value + 0.1)
    state.value = 'ZOOMING'
  }
  
  const zoomOut = () => {
    scale.value = Math.max(0.5, scale.value - 0.1)
    state.value = 'ZOOMING'
  }
  
  const resetZoom = () => {
    scale.value = 1
    position.value = { x: 0, y: 0 }
    state.value = 'READING'
  }
  
  // 切换模式
  const switchMode = async () => {
    const newMode = pageMode.value === 'left_right' ? 'up_down' : 'left_right'
    configStore.setPageMode(newMode)
    await saveProgress()
  }
  
  // 切换背景
  const switchBackground = async () => {
    const backgrounds = ['white', 'dark', 'eye_protection']
    const currentIndex = backgrounds.indexOf(background.value)
    const nextIndex = (currentIndex + 1) % backgrounds.length
    configStore.setBackground(backgrounds[nextIndex])
    await saveProgress()
  }
  
  // 键盘事件
  const handleKeydown = (event) => {
    const keyMap = {
      'ArrowLeft': prevPage,
      'a': prevPage,
      'A': prevPage,
      'ArrowRight': nextPage,
      'd': nextPage,
      'D': nextPage,
      ' ': nextPage,
      'ArrowUp': () => pageMode.value === 'up_down' && prevPage(),
      'w': () => pageMode.value === 'up_down' && prevPage(),
      'W': () => pageMode.value === 'up_down' && prevPage(),
      'ArrowDown': () => pageMode.value === 'up_down' && nextPage(),
      's': () => pageMode.value === 'up_down' && nextPage(),
      'S': () => pageMode.value === 'up_down' && nextPage(),
      '+': zoomIn,
      '=': zoomIn,
      '-': zoomOut,
      '_': zoomOut,
      '0': resetZoom,
      'b': switchBackground,
      'B': switchBackground,
      'm': switchMode,
      'M': switchMode,
      'Escape': () => router.back()
    }
    
    const handler = keyMap[event.key]
    if (handler) {
      event.preventDefault()
      handler()
    }
  }
  
  return {
    // 状态
    state,
    currentPage,
    scale,
    position,
    // 配置
    pageMode,
    background,
    preloadCount,
    // 计算属性
    progress,
    canGoNext,
    canGoPrev,
    // 方法
    loadComic,
    nextPage,
    prevPage,
    jumpToPage,
    zoomIn,
    zoomOut,
    resetZoom,
    switchMode,
    switchBackground,
    handleKeydown
  }
}
```

```javascript
// composables/useImport.js
import { ref } from 'vue'
import { useImportStore } from '@/stores/import'

export function useImport() {
  const importStore = useImportStore()
  
  const loading = ref(false)
  const error = ref(null)
  const progress = ref(0)

        comic_id: comicId
      })

      return result
    } catch (err) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }
  
  // 目录导入

    loading.value = true
    error.value = null
    progress.value = 0
    
    try {
      const result = await importStore.importByDirectory({
        source_dir: sourceDir,
        comic_id: comicId,
        title: title
      })
      
      progress.value = 100
      return result
    } catch (err) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }

      const formData = new FormData()

      formData.append('comic_id', comicId)
      formData.append('title', title)
      
      const result = await importStore.importByZip(formData)
      
      progress.value = 100
      return result
    } catch (err) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }

        image_urls: imageUrls
      })
      
      progress.value = 100
      return result
    } catch (err) {
      error.value = err.message
      throw err
    } finally {
      loading.value = false
    }
  }
  
  return {
    loading,
```

#### 3.3.3 阅读器组件拆分

```vue
<!-- components/reader/ImageViewer.vue -->
<template>
  <div 
    class="image-viewer"
    :class="[`mode-${mode}`, `bg-${background}`]"
    @click="handleClick"
  >
    <!-- 左右翻页模式 -->
    <template v-if="mode === 'left_right'">
      <img
        :src="currentImageUrl"
        :style="imageStyle"
        @load="handleLoad"
        @error="handleError"
      />
    </template>
    
    <!-- 上下翻页模式 -->
    <template v-else>
      <div class="vertical-images" ref="containerRef">
        <img
          v-for="page in visiblePages"
          :key="page"
          :src="getImageUrl(page)"
          :data-page="page"
          @load="handleLoad"
          @error="handleError"
        />
      </div>
    </template>
  </div>
</template>

<script setup>
import { computed, ref } from 'vue'

const props = defineProps({
  comicId: String,
  currentPage: Number,
  totalPage: Number,
  mode: String,
  background: String,
  scale: Number,
  position: Object
})

const emit = defineEmits(['load', 'error', 'click'])

const containerRef = ref(null)

// 当前图片 URL
const currentImageUrl = computed(() => {
  return `/api/v1/image?comic_id=${props.comicId}&page_num=${props.currentPage}`
})

// 可见页码（上下模式）
const visiblePages = computed(() => {
  if (props.mode !== 'up_down') return []
  
  // 显示当前页前后各 3 页
  const pages = []
  for (let i = -3; i <= 3; i++) {
    const page = props.currentPage + i
    if (page >= 1 && page <= props.totalPage) {
      pages.push(page)
    }
  }
  return pages.sort((a, b) => a - b)
})

// 图片样式
const imageStyle = computed(() => ({
  transform: `scale(${props.scale}) translate(${props.position.x}px, ${props.position.y}px)`,
  transition: 'transform 0.2s ease'
}))

// 获取图片 URL
const getImageUrl = (page) => {
  return `/api/v1/image?comic_id=${props.comicId}&page_num=${page}`
}

const handleLoad = () => emit('load')
const handleError = () => emit('error')
const handleClick = (e) => emit('click', e)
</script>
```

### 4.2 应对策略

1. **逐步重构**：不一次性重构所有代码，分模块逐步替换
2. **保留旧代码**：新旧代码并存，验证通过后再删除旧代码
3. **完善测试**：重构前编写完整的测试用例，确保功能不变
4. **代码审查**：关键重构点进行代码审查，确保质量
5. **文档同步**：重构同时更新文档，降低学习成本
